    // Helper function to validate email

    function showAlert(message, type = 'info') {

        const modalContent = document.querySelector('.modal-content');

            });

    <script>
        // Função para buscar imóveis
        (function() {
            const searchButton = document.getElementById('search-button');
            const locationInput = document.getElementById('search-location');
            const priceInput = document.getElementById('search-price');
            const bedroomsInput = document.getElementById('search-bedrooms');
            const cards = document.querySelectorAll('.listing-card');
            const noResultsMessage = document.getElementById('no-results-message');

            function parsePriceRange(value) {
                if (!value) return { min: null, max: null };
                const numbers = (value.match(/\d+/g) || []).map(Number);
                if (numbers.length === 1) {
                    return { min: 0, max: numbers[0] };
                }
                if (numbers.length >= 2) {
                    return { min: numbers[0], max: numbers[1] };
                }
                return { min: null, max: null };
            }

            function applyFilters() {
                const locationValue = (locationInput.value || '').toLowerCase().trim();
                const { min, max } = parsePriceRange(priceInput.value);
                const bedroomsValue = parseInt(bedroomsInput.value, 10);

                let anyVisible = false;

                cards.forEach(card => {
                    const city = (card.getAttribute('data-city') || '').toLowerCase();
                    const price = parseInt(card.getAttribute('data-price') || '0', 10);
                    const bedrooms = parseInt(card.getAttribute('data-bedrooms') || '0', 10);

                    let visible = true;

                    if (locationValue && !city.includes(locationValue)) {
                        visible = false;
                    }

                    if (visible && (min !== null || max !== null)) {
                        if (min !== null && price < min) visible = false;
                        if (max !== null && price > max) visible = false;
                    }

                    if (visible && !Number.isNaN(bedroomsValue)) {
                        if (bedrooms < bedroomsValue) visible = false;
                    }

                    card.style.display = visible ? '' : 'none';
                    if (visible) anyVisible = true;
                });

                if (!anyVisible) {
                    noResultsMessage.style.display = 'block';
                } else {
                    noResultsMessage.style.display = 'none';
                }
            }

            if (searchButton) {
                searchButton.addEventListener('click', function (e) {
                    e.preventDefault();
                    applyFilters();
                });
            }
        })();
    </script>
